Subarray Sum I

9 Sep, 2015 · Read in about 1 min · (105 Words)
Tags: Prefix Sum; Sort; Subarray;          Topics: Algorithm;

Given an integer array, find a subarray where the sum of numbers is zero. Your code should return the index of the first number and the index of the last number.

    
public class Solution {
    /**
     * @param nums: A list of integers
     * @return: A list of integers includes the index of the first number 
     *          and the index of the last number
     */
    public ArrayList subarraySum(int[] nums) {
        // write your code here
        ArrayList res = new ArrayList();
        if(nums == null || nums.length == 0)
            return res;       
        HashMap map = new HashMap();
        int sum = 0;
        map.put(0, 0);
        for(int i = 0; i < nums.length; i++) {
            if(map.containsKey(sum+nums[i])) {
                res.add(map.get(sum+nums[i]));
                res.add(i);
                return res;
            }
            map.put(sum+=nums[i], i+1);
        }
        return res;
    }
}
